college = []
city = []
Address = []
established  = []
Ownership = []
Affiliated = []

for i in range(1,24):
    url = f'https://www.indiaccess.com/colleges/medical/?currentpage={i}'
    
    headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120 Safari/537.36"
    }
    page = requests.get(url ,headers=headers)
    soup = BeautifulSoup(page.text)
    brand_tag= soup.find_all('h4')

    for i in brand_tag:
        a = (i.text)
        college.append(a.split(',')[0])

    for i in brand_tag:
        a = i.text.replace('\n','')
        city.append(a.split(',')[1])    

    address = soup.find_all('div',class_="additional-info")

    for i in address:
        a = i.text
        b = i.get_text(strip = True)
        c = i.text.replace('\n',' ')
        Address.append(c)
    
    year = soup.find_all('span',style="font-size:12px; margin-right:10px;")

    for i in year:
        b=i.text
        a = re.findall(r'Established:\s(\d{4})',b)
        established.extend(a)    

    for i in year:
        b=i.text
        a = re.findall(r'Ownership:\s(\w+)',b)
        Ownership.extend(a)    

    for i in year:
        b=i.text
        a = re.findall(r'Affiliated to:\s(.+)',b)
        Affiliated.extend(a)

college, city, Address, established, Ownership, Affiliated = zip(*zip(college, city, Address, established, Ownership, Affiliated)) if college else ([], [], [], [], [], [])
